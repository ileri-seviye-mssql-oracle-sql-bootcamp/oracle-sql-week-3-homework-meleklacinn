            
---firma---
CREATE TABLE FIRMA.firma
  (
    f_Id NUMBER GENERATED BY DEFAULT AS IDENTITY
    START WITH 5 PRIMARY KEY,
    firma_adi VARCHAR2( 50 ) NOT NULL,
    
     firma_sahibi VARCHAR2( 50 ) NOT NULL
 
        
  );
---proje---
CREATE TABLE FIRMA.proje
  (
    p_Id NUMBER GENERATED BY DEFAULT AS IDENTITY
    START WITH 5 PRIMARY KEY,
    proje_adi VARCHAR2( 50 ) NOT NULL,
     proje_sure VARCHAR2( 50 ) NOT NULL,
     proje_butce VARCHAR2( 50 ) NOT NULL,
      proje_eleman VARCHAR2( 10 ) NOT NULL,
       CONSTRAINT fk_proje_firma FOREIGN KEY( f_Id)
      REFERENCES FIRMA.firma( f_Id )
      ON DELETE CASCADE);
---developer---
CREATE TABLE FIRMA.developer
  (
    d_Id NUMBER GENERATED BY DEFAULT AS IDENTITY
    START WITH 5 PRIMARY KEY,
    yazilimci_adi VARCHAR2( 50 ) NOT NULL
    
       CONSTRAINT fk_developer_firma FOREIGN KEY( f_Id)
      REFERENCES FIRMA.firma( f_Id )
      ON DELETE CASCADE,
      CONSTRAINT fk_developer_proje 
      FOREIGN KEY( pId )
      REFERENCES FIRMA.PROJE(pId)
      ON DELETE SET NULL
    
  );
